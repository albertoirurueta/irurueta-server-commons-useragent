16  package com.irurueta.server.commons.useragent;
17  
18  /**
19   * This enum represents the type of a user agent. 
20   */
21  public enum UserAgentType {
22      /**
23       * A web browser, especially on a desktop, notebook or workstation.
24       */
25      BROWSER,
26      
27      /**
28       * An email client, email reader.
29       */
30      EMAIL_CLIENT,
31      
32      /**
33       * A news aggregator, also termed as feed aggregator, feed reader, news 
34       * reader or RSS reader.
35       */
36      FEED_READER,
37      
38      /**
39       * A library is a collection of resources used to develop software.
40       */
41      LIBRARY,
42      
43      /**
44       * Media player, also called multimedia player, is a term typically used to 
45       * describe computer software for playing back multimedia files.
46       */
47      MEDIAPLAYER,
48      
49      /**
50       * A mobile browser, also called a microbrowser, minibrowser, or wireless
51       * internet browser (WIB), is a web browser designed for use on a mobile
52       * device.
53       */
54      MOBILE_BROWSER,
55      
56      /**
57       * Offline browser which may download completely or partially a website to a
58       * hard drive.
59       */
60      OFFLINE_BROWSER,
61      
62      /**
63       * Bots, such as Web crawlers.
64       */
65      ROBOT,
66      
67      /**
68       * A software to hide the real user agent information.
69       */
70      USERAGENT_ANONYMIZER,
71      
72      /**
73       * A software or service that validates parts of a website or webservice, 
74       * such as CSS, HTML, JSON, XML.
75       */
76      VALIDATOR,
77      
78      /**
79       * A WAP browser is a web browser for mobile devices such as mobile phones 
80       * that use the Wireless Application Protocol (WAP).
81       */
82      WAP_BROWSER,
83      
84      /**
85       * A software that doesn't match the other types.
86       */
87      OTHER,
88      
89      /**
90       * An unknown user agent type.
91       */
92      UNKNOWN
93  }
