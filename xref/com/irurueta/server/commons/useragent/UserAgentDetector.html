31  /**
32   * Class to parse and process a user agent from a web browser, web crawler, bot, email client, RESt library, etc. This class will detect
33   * which kind of user agent made a request, operating system information of the requester, kind of device, etc.
34   */
35  public class UserAgentDetector implements Closeable {
36  
37    /**
38     * Logger for this class.
39     */
40    private static final Logger LOG = Logger.getLogger(UserAgentDetector.class.
41        getName());
42  
43    /**
44     * Singleton instance of UserAgentDetector stored in a soft reference (to keep it cached in memory unless memory is claimed).
45     */
46    private static SoftReference<UserAgentDetector> mReference;
47  
48    /**
49     * Indicates whether user agent detection is enabled or not.
50     */
51    private boolean mEnabled;
52  
53    /**
54     * Amount of user agents that are cached.
55     */
56    private int mCacheSize;
57  
58    /**
59     * Amount of time to keep user agents cached expressed in hours.
60     */
61    private int mCacheExpirationTime;
62  
63    /**
64     * Internal user agent string parser.
65     */
66    private UserAgentStringParser mParser;
67  
68    /**
69     * Cache to hold instantiated instances of radable user agents.
70     */
71    private Cache<String, ReadableUserAgent> mCache;
72  
73    /**
74     * Constructor. Creates and configures a UserAgentDetector instance.
75     */
76    private UserAgentDetector() {
77      mEnabled = false;
78      try {
79        final UserAgentConfiguration cfg = UserAgentConfigurationFactory.
80            getInstance().configure();
81        mEnabled = cfg.isUserAgentDetectionEnabled();
82        if (mEnabled) {
83          mCacheSize = cfg.getUserAgentCacheSize();
84          mCacheExpirationTime =
85              cfg.getUserAgentCacheExpirationTimeHours();
86  
87          mParser = UADetectorServiceFactory.getCachingAndUpdatingParser();
88          mCache = CacheBuilder.newBuilder().maximumSize(mCacheSize).
89              expireAfterWrite(mCacheExpirationTime, TimeUnit.HOURS).
90              build();
91          LOG.log(Level.INFO, "User agent detection is enabled");
92        } else {
93          LOG.log(Level.INFO, "User agent detection is disabled");
94        }
95  
96      } catch (Exception e) {
97        mEnabled = false;
98        LOG.log(Level.INFO, "User agent detection is disabled because " +
99            "configuration failed", e);
100     }
101   }
102 
103   /**
104    * Factory method to return the singleton instance of UserAgentDetector based on current configuration.
105    *
106    * @return singleton instance.
107    */
108   public static synchronized UserAgentDetector getInstance() {
109     UserAgentDetector detector;
110     if (mReference == null || (detector = mReference.get()) == null) {
111       detector = new UserAgentDetector();
112       mReference = new SoftReference<>(detector);
113     }
114     return detector;
115   }

118    * Indicates whether user agent detection is enabled or not. If not enabled no user agent detection will be done when requested.
119    *
120    * @return true if user agent detection is enabled, false otherwise.

122   public boolean isEnabled() {

145   /**
146    * Detects data on provided user agent string. Detected data can be operating system, user agent type (browser, mail client, etc), user
147    * agent family, type of device, etc.
148    *
149    * @param userAgentString original user agent string being parsed
150    * @return detected user agent data.
151    * @throws UserAgentDetectionDisabledException if user agent detection is disabled.
152    * @throws UserAgentException if anything else fails.
153    */
154   public UserAgentData detect(final String userAgentString) throws
155       UserAgentDetectionDisabledException, UserAgentException {
156     if (!mEnabled) {
157       throw new UserAgentDetectionDisabledException();
158     }
159 
160     try {
161       ReadableUserAgent result = mCache.getIfPresent(userAgentString);
162       if (result == null) {
163         result = mParser.parse(userAgentString);
164         mCache.put(userAgentString, result);
165       }
166 
167       final DeviceCategory deviceCategory = toDeviceCategory(
168           result.getDeviceCategory().getCategory());
169       final String deviceCategoryName = result.getDeviceCategory().getName();
170       final String family = result.getFamily().getName();
171       final OperatingSystemFamily osFamily = toOsFamily(result.getOperatingSystem().getFamily());
172       final String osFamilyName = result.getOperatingSystem().getFamilyName();
173       final String osName = result.getOperatingSystem().getName();
174       final String osProducer = result.getOperatingSystem().getProducer();
175       final String osVersion = result.getOperatingSystem().getVersionNumber().toVersionString();
176       final UserAgentType userAgentType = toUserAgentType(result.getType());
177       final String userAgentVersion = result.getVersionNumber().toVersionString();
178 
179       return new UserAgentData(userAgentString, deviceCategory,
180           deviceCategoryName, family, osFamily, osFamilyName, osName,
181           osProducer, osVersion, userAgentType, userAgentVersion);
182     } catch (final Exception e) {
183       throw new UserAgentException(e);
184     }
185   }
186 
187   /**
188    * Stops internal user agent parser. Once closed, user agent detection will no longer be available.
189    */
190   @Override
191   public void close() {
192     if (mParser != null) {
193       mParser.shutdown();
194     }
195     if (mCache != null) {
196       mCache.invalidateAll();
197     }
198     mEnabled = false;
199     LOG.log(Level.INFO, "User agent detection has been shutdown");
200   }
201 
202   /**
203    * Resets UserAgentDetector so a new instance having new configuration can be created.
204    */
205   protected static synchronized void reset() {
206     UserAgentDetector detector = mReference != null ?
207         mReference.get() : null;
208     if (detector != null) {
209       detector.close();
210     }
211     mReference = null;
212   }

214   /**
215    * This method is called on garbage collection. When this method is called, internal user agent parser will be stopped.
216    *
217    * @throws Throwable if anything fails.
218    */
219   @Override
220   protected void finalize() throws Throwable {
221     try {
222       close();
223     } finally {
224       super.finalize();
225     }
226   }

228   /**
229    * Converts an internal device category enumerator into a DeviceCategory enumerator used by this package.
230    *
231    * @param category internal category to be converted.
232    * @return a device category.
233    */
234   protected DeviceCategory toDeviceCategory(
235       ReadableDeviceCategory.Category category) {
236     if (category != null) {
237       switch (category) {
238         case GAME_CONSOLE:

242         case PDA:
243           return DeviceCategory.PDA;
244         case PERSONAL_COMPUTER:
245           return DeviceCategory.PERSONAL_COMPUTER;
246         case SMARTPHONE:
247           return DeviceCategory.SMARTPHONE;
248         case SMART_TV:
249           return DeviceCategory.SMART_TV;
250         case TABLET:
251           return DeviceCategory.TABLET;
252         case UNKNOWN:
253           return DeviceCategory.UNKNOWN;
254         case WEARABLE_COMPUTER:
255           return DeviceCategory.WEARABLE_COMPUTER;
256         default:
257           break;
258       }
259     }
260     return null;
261   }
